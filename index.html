Coder en Python

app/main.py : 
# Imports
import os
import sys
from app.functions import convert

# Fonction principale
def main():
	print("Bienvenue !")
	age_utilisateur = input("Entrez votre age : ")
	categorie = convert(age_utilisateur)
	print(f"Vous appartenenez aux {categorie} ans !")

# Execution du code
if __name__ == "__main__":
	main()
